87:    public ArrayDeque shunting(ArrayDeque<Token> infix) throws IOException{
88:        operatorStack = new Stack<Character>();
89:        postfixDeque = new ArrayDeque<String>();
90:        while (!infix.isEmpty()){
91:            Token token = infix.poll();
92:            if (token.type == "OPERATOR") { 
93:                processOperator((token.data).charAt(0));
94:            }else if(token.type == "SIN"){
95:                // TODO: This is kind of a weird place to put this... should be delegated to a more generic method.
96:                String d = "";
97:                Pattern pattern = Pattern.compile("([-]?[0-9.]+)");
98:                Matcher matcher = pattern.matcher(token.data);
99:                while (matcher.find()) {
100:                    d = matcher.group(1);
101:                }
102:                processOperator('s'); // 's' will be sin function.
103:                postfixDeque.add(d);
104:            }else{
105:                postfixDeque.add(token.data);
106:            }
107:        }
108:        // Pop any remaining operators and
109:        // append them to postfix.
110:        while (!operatorStack.empty()) {
111:            char op = operatorStack.pop();
112:            postfixDeque.add(String.valueOf(op));
113:        }
114:        return postfixDeque;
115:    }
116:    private void processOperator(char op){
117:        if (operatorStack.empty()) {
118:            operatorStack.push(op);
119:        } else {
120:            char topOp = operatorStack.peek();
121:            if(op == '('){
122:                operatorStack.push(op);
123:            }else if ((topOp == '(' || precedence(op) > precedence(topOp)) && op != ')') {
124:                operatorStack.push(op);
125:            }else if(op == ')'){
126:                while(topOp != '('){
127:                    operatorStack.pop();
128:                    postfixDeque.add(String.valueOf(topOp));
129:                    if (!operatorStack.empty()) {
130:                        topOp = operatorStack.peek();
131:                    }
132:                }
133:                operatorStack.pop(); // removes the right parenthesis
134:            } else {
135:                // Pop all stacked operators with equal
136:                // or higher precedence than op.
137:                while (!operatorStack.empty() && precedence(op) <= precedence(topOp) && topOp != '(') {
138:                    operatorStack.pop();
139:                    postfixDeque.add(String.valueOf(topOp));
140:                    if (!operatorStack.empty()) {
141:                        // Reset topOp.
142:                        topOp = operatorStack.peek();
143:                    }
144:                }
145:                operatorStack.push(op);
146:            }
147:        }

150:         double result = 0.0;
151:         double first = 0.0;
152:         double second = 0.0;
153:         operandStack = new Stack<Double>();
154:         while(!postfix.isEmpty()) {
