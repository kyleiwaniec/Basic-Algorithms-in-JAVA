1:import java.util.ArrayList;
2:import java.util.List;
3:public class PrimeFactors {
4:  public static List<Integer> primeFactors(int numbers) {
5:    int n = numbers;
6:    List<Integer> factors = new ArrayList<Integer>();
7:    for (int i = 2; i <= n / i; i++) {
8:      while (n % i == 0) {
9:        factors.add(i);
10:        n /= i;
11:      }
12:    }
13:    if (n > 1) {
14:      factors.add(n);
15:    }
16:    return factors;
17:  }
18:  public static void main(String[] args) {
19:    int n = Integer.parseInt(args[0]);
20:    System.out.println("Primefactors of "+n);

22:      System.out.println(integer);
23:    }
24:  }
25:}
